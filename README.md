# GUIExample

import java.awt.*;
public class GUIExample extends JFrame {
public GUIExample() {
setTitle("My First GUI");
setSize(500,400);
setVisible(true);
setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
}
public static void main(String [] args) {
GUIExample = gui = new GUIExample();
}
}
